    **parsePeriod 헬퍼 함수 (컴포넌트 외부에 정의):**
    - 유효한 PeriodKey 목록에 raw 값이 있으면 반환, 없으면 탭 기본값 반환
    - Daily 유효 목록: ['this-week', 'last-week', 'this-month', 'last-month']
    - Weekly 유효 목록: ['this-month', 'last-month']

    **DashboardContent 컴포넌트:**
    - useState로 period 상태 관리 (초기값: parsePeriod(initialPeriod, tab))
    - useCallback으로 handlePeriodChange 정의:
      - setPeriodState(newPeriod)
      - `const params = new URLSearchParams(searchParams.toString())`
      - `params.set('period', newPeriod)`
      - `router.replace(\`\${pathname}?\${params.toString()}\`, { scroll: false })`
    - useMemo로 filteredData 계산 (의존성: [data, tab, period]):
      - tab === 'daily': filterDailyByPeriod(data.daily, getDateRange(period)) → { ...data, daily: filtered }
      - tab === 'weekly': filterWeeklyByPeriod(data.weekly, period as 'this-month' | 'last-month') → { ...data, weekly: filtered }
    - 렌더링:
      ```tsx
      <div className="space-y-6">
        <DashboardHeader tab={tab} period={period} onPeriodChange={handlePeriodChange} />
        <KpiCards data={filteredData} tab={tab} />
        <ChartsSection data={filteredData} tab={tab} />
        <DataTable data={filteredData} tab={tab} />
      </div>
      ```

    **최종 page.tsx 구조:**
    ```tsx
    import { Suspense } from 'react';
    import { getTeamDashboardData } from '@/lib/data';
    import { DashboardContent } from '@/components/dashboard/dashboard-content';
    import { KpiCardsSkeleton } from '@/components/dashboard/kpi-cards-skeleton';
    import { UpdateTimestamp } from '@/components/dashboard/update-timestamp';

    export const dynamic = 'force-dynamic';
    type SearchParams = Promise<{ tab?: string; period?: string }>;

    export default async function DashboardPage({ searchParams }: { searchParams: SearchParams }) {
      const { tab = 'daily', period } = await searchParams;
      const activeTab = tab === 'weekly' ? 'weekly' : 'daily';
      const data = await getTeamDashboardData();

      return (
        <div className="space-y-6">
          <div className="flex justify-end">
            <UpdateTimestamp fetchedAt={data.fetchedAt} />
          </div>
          <Suspense fallback={<KpiCardsSkeleton />}>
            <DashboardContent data={data} tab={activeTab} initialPeriod={period} />
          </Suspense>
        </div>
      );
    }
    ```
